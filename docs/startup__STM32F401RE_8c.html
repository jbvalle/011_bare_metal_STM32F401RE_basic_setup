void Reset_handler(void){
uint32_t * pSRC = (uint32_t *)&_sidata;
uint32_t * pDST = (uint32_t *)&_sdata;
for(uint32_t *dataptr = (uint32_t *)pDST; dataptr < &_edata;){
*dataptr++ = *pSRC++;
}
for(uint32_t *bss_ptr = (uint32_t *)&_sbss; bss_ptr < &_ebss;){
*bss_ptr++ = 0;
}
main();
}
